This is a simple HTML page with a Tcl script that demonstrates the use of the Sample command.

Sample

 
    
# open and close
set fileHandle [open "filename.txt" mode]
# mode "r" (read), "w" (write), "a" (append), or "b" (binary mode)
# Perform operations on the file
close $fileHandle
    


 
    
# reading file 

set fileHandle [open "filename.txt" "r"]
while {[gets $fileHandle line] >= 0} {
    # Process each line
    puts $line
}
close $fileHandle

# Alternatively, you can use the read command to read the entire contents of a file into a variable:
set fileHandle [open "filename.txt" "r"]
set fileContents [read $fileHandle]
close $fileHandle
    


 
    
# writing file

set fileHandle [open "output.txt" "w"]
puts $fileHandle "This is a line of text."
close $fileHandle
    


 
    
set fileHandle [open "existing_file.txt" "a"]
puts $fileHandle "This data will be appended."
close $fileHandle
