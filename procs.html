# Define a procedure named "sum_numbers"
proc sum_numbers {num1 num2} {
    # Calculate the sum of num1 and num2
    set result [expr $num1 + $num2]
    return $result
}

# Call the procedure with two numbers
set number1 5
set number2 7
set sum_result [sum_numbers $number1 $number2]

# Display the result
puts "The sum of $number1 and $number2 is: $sum_result"
